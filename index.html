});
const questions = [
{
q: "문제 1: 시리아는 현재 휴전 상태이다. (O/X)",
correct: false,
correctText: "정답입니다! 시리아는 여전히 분쟁 중입니다.",
wrongText: "오답입니다. 시리아는 아직 휴전하지 않았습니다."
},
{
q: "문제 2: 우크라이나는 현재 전쟁 중이다. (O/X)",
correct: true,
correctText: "정답입니다! 현재 우크라이나는 전쟁 중입니다.",
wrongText: "오답입니다. 우크라이나는 아직 전쟁 중입니다."
},
{
q: "문제 3: 나이지리아는 테러 위험 지역이다. (O/X)",
correct: true,
correctText: "정답입니다! 나이지리아는 테러 위험이 존재합니다.",
wrongText: "오답입니다. 나이지리아는 여전히 위험 지역입니다."
}
];
let current = 0;
let score = 0;
function submitAnswer(userAnswer) {
const result = document.getElementById("quiz-result");
const q = questions[current];
if (userAnswer === q.correct) {
score++;
result.textContent = q.correctText;
result.style.color = "#00ffcc";
} else {
result.textContent = q.wrongText;
result.style.color = "#ff6666";
}
current++;
if (current < questions.length) {
setTimeout(() => {
result.textContent = "";
document.getElementById("quiz-question").textContent = questions[current].q;
}, 1500);
} else {
setTimeout(() => {
let comment = "";
if (score === 3) comment = "축하합니다!";
else if (score === 2) comment = "잘했어요!";
else comment = "더 알아볼까요?";
document.getElementById("quiz-section").innerHTML =
"<h3>퀴즈 완료!</h3><p>총 점수: " + score + " / 3점</p><p>" + comment + "</p>";
}, 2000);
}
}
